30      private final Map<Class, Map<String, Object>> createdEntities = new LinkedHashMap<>();
31      private final MissingPropertiesReport report = new MissingPropertiesReport();
32      private final Object unitTest;
33      private final Set<Client> clients;
34      private final Set<DataSetMappingDefinition.WhenLoaded> whenLoadedListeners = new LinkedHashSet<>();
35      private final Set<DataSetMappingDefinition.OnChange> onChangeListeners = new LinkedHashSet<>();
36  
37      DefaultDataSet(Object unitTest, Set<Client> clients) {
38          this.unitTest = unitTest;
39          this.clients = clients;
40      }

42      @Override
43      @SuppressWarnings("unchecked")
44      public <T> T findByTypeAndOriginalId(Class<T> type, Object originalId) {
45          Map<String, Object> instancesByType = createdEntities.get(type);
46          if (instancesByType == null) {
47              return null;
48          }
49          return (T) instancesByType.get(String.valueOf(originalId));
50      }
51  
52      @Override
53      @SuppressWarnings("unchecked")
54      public <T> List<T> findAllByType(Class<T> type) {
55          Map<String, Object> instancesByType = createdEntities.get(type);
56          if (instancesByType == null) {
57              return Collections.emptyList();
58          }
59          return Collections.unmodifiableList(new ArrayList<>((Collection<T>) instancesByType.values()));
60      }
61  
62      @Override
63      public <T> T findByType(Class<T> type) {
64          List<T> ret = findAllByType(type);
65          if (ret != null && !ret.isEmpty()) {
66              return ret.get(0);
67          }
68          return null;
69      }
70  
71      @Override
72      public <T> T add(T entity) {
73          return add(entity, null);
74      }
75  
76      @Override
77      public <T> T add(T entity, Object manualId) {
78          Map<String, Object> instancesByType = createdEntities.get(entity.getClass());
79          if (instancesByType == null) {
80              instancesByType = new LinkedHashMap<>();
81              createdEntities.put(entity.getClass(), instancesByType);
82          }
83  
84          String id = manualId != null ? String.valueOf(manualId) : findClient(entity.getClass()).getId(entity);
85          String systemId = String.valueOf(System.identityHashCode(entity));
86  
87          if (id == null) {
88              id = systemId;
89          }
90  
91          if (!systemId.equals(id)) {
92              instancesByType.remove(systemId);
93              instancesByType.put(id, entity);
94          } else {
95              instancesByType.put(systemId, entity);
96          }
97  
98          changed();
99  
100         return entity;
101     }
102 
103     @Override
104     @SuppressWarnings("unchecked")
105     public <T> T remove(T object) {
106         Map<String, Object> instances = createdEntities.get(object.getClass());
107         if (instances == null) {
108             return null;
109         }
110 
111         String key = findClient(object.getClass()).getId(object);
112 
113         T removed = (T) instances.remove(key);
114 
115         if (removed != null) {
116 
117             changed();
118 
119             return removed;
120         }
121 
122         for (Map.Entry<String, Object> entry : instances.entrySet()) {
123             if (entry.getValue() != null && entry.getValue().equals(object)) {
124                 key = entry.getKey();
125                 break;
126             }
127         }
128 
129         removed = key != null ? (T) instances.remove(key) : null;
130 
131         changed();
132 
133         return removed;
134     }

148     /**
149      * Signals that data sets was manually loaded or the data has been changed significantly.
150      * @return self
151      */
152     @Override
153     public DataSet loaded() {
154         for (DataSetMappingDefinition.WhenLoaded whenLoaded : whenLoadedListeners) {
155             whenLoaded.doWhenLoaded(this);
156         }
157         return this;
158     }
159 
160     /**
161      * Signals that data sets was changed outside the dataset.
162      * @return self
163      */
164     @Override
165     public DataSet changed() {
166         for (DataSetMappingDefinition.OnChange onChange : onChangeListeners) {
167             onChange.doOnChange(this);
168         }
169         return this;
170     }

172     @Override
173     public DataSet load(Class<?> self, Consumer<DataSetMappingDefinition> configuration) {
174         return load(new PreparedDataSet(self, configuration));

177     DataSet load(DataSetMapping mapping) {
178         loadInternal(mapping);
179         return loaded();
180     }

182     @Override
183     public MissingPropertiesReport getReport() {
184         return report;

187     private void loadInternal(DataSetMapping mapping) {
188         for (Source source : mapping.getSources().values()) {
189             Parser parser = Parsers.findParser(source);
190             Object content = parser.getContent(source);
191             for (PropertyMapping propertyMapping : source.getRootPropertyMappings()) {
192                 String path = propertyMapping.getPath();
193                 Iterable<Map<String, Object>> matching = parser.findAllMatching(content, path);
194                 for (Map<String, Object> match : matching) {
195                     propertyMapping.processPropertyValue(this, mapping, parser, null, match);
196                 }
197             }
198         }
199         whenLoadedListeners.addAll(mapping.getWhenLoadedListeners());
200         onChangeListeners.addAll(mapping.getOnChangeListeners());
201     }
202 
203     private Client findClient(Class type) {
204         Client result = null;
205         for (Client client: clients) {
206             if (client.isSupported(type)) {
207                 result = client;
208                 break;
209             }
210         }
211         return Preconditions.checkNotNull(result, "Client not found for " + type);
212     }
