114     @Override
115     public <T> T newInstance(Parser parser, Class<T> type, Map<String, Object> payload) {
116         BeanIntrospection<T> introspection = ops().getEntity(type).getIntrospection();
117         T instance = introspection.instantiate(
118             Arrays.stream(introspection.getConstructorArguments()).map(Argument::getName).map(payload::get).toArray()
119         );
120         payload.forEach((k, v) -> introspection.getProperty(k).ifPresent(p -> {
121             if (!p.isReadOnly()) {
122                 p.set(instance, v);
123             }
124         }));
125         return instance;
126     }

128     @Override
129     @SuppressWarnings({"rawtypes", "unchecked"})
130     public <T> T save(T object) {
131         Class<?> type = object.getClass();
132         ApplicationContext ctx = provider.getApplicationContext();
133 
134         SynchronousTransactionManager<T> transactionManager = ctx.getBean(SynchronousTransactionManager.class);
135         Collection<BeanDefinition<GenericRepository>> definitions = ctx.getBeanDefinitions(GenericRepository.class);
136 
137         for (BeanDefinition<GenericRepository> definition : definitions) {
138             Optional<ExecutableMethod<GenericRepository, ?>> maybeSaveMethod = tryFindSaveMethod(type, definition);
139 
140             if (maybeSaveMethod.isPresent()) {
141                 GenericRepository repository = ctx.getBean(definition.getBeanType());
142 
143                 Object id = getOriginalId(object);
144 
145                 try {
146                     if (id != null) {
147                         Optional<ExecutableMethod<GenericRepository, ?>> maybeFindByIdMethod = tryFindFindByIdMethod(type, definition);
148 
149                         if (!maybeFindByIdMethod.isPresent()) {
150                             LOGGER.warn(
151                                 "Skipping object " + object + " with id " + id + " which is probably already persisted as there is no easy way how to update it. Please, provide 'findById' in " + repository.getClass() + " class: "
152                                     + object
153                                     + "! If you are missing some information then re-arrange the fixtures so the most complete fixture goes first."
154                             );
155                             return object;
156                         }
157 
158                         Object existing = transactionManager.executeRead(s -> maybeFindByIdMethod.get().invoke(repository, getOriginalId(object)));
159                         Maybe<T> maybeExists = asMaybeEntity(existing, object);
160                         if (!maybeExists.isEmpty().blockingGet()) {
161                             Optional<ExecutableMethod<GenericRepository, ?>> maybeUpdateMethod = tryFindUpdateMethod(type, definition);
162                             if (maybeUpdateMethod.isPresent()) {
163                                 T updated = asMaybeEntity(transactionManager.executeWrite(s -> maybeUpdateMethod.get().invoke(repository, object)), object)
164                                     .doOnError(th -> LOGGER.error("Exception saving object asynchronously " + object, th))
165                                     .blockingGet(object);
166                                 LOGGER.debug("Updated object " + object + " as " + updated);
167                                 return updated;
168                             }
169                             LOGGER.warn(
170                                 "Skipping object " + object + " with id " + id + " which is probably already persisted as there is no easy way how to update it.Please, provide 'update' method in  " + repository.getClass() + "  class: "
171                                     + object
172                                     + "! If you are missing some information then re-arrange the fixtures so the most complete fixture goes first."
173                             );
174                             return object;
175                         }
176                     }
177                 } catch (Exception e) {
178                     LOGGER.warn(
179                         "Skipping object  " + object + " with id " + id + " which is probably already persisted as there were issues during the update! " +
180                             "If you are missing some information then re-arrange the fixtures so the most complete fixture goes first.",
181                         e
182                     );
183                     return object;
184                 }
185 
186                 try {
187                     ExecutableMethod<GenericRepository, ?> saveMethod = maybeSaveMethod.get();
188                     if (!definition.findAnnotation("io.micronaut.data.jdbc.annotation.JdbcRepository").isPresent()) {
189                         // JPA does not support id set on a new object
190                         removeId(object);
191                     }
192                     Object result = transactionManager.executeWrite(s -> saveMethod.invoke(repository, object));
193                     T saved = asMaybeEntity(result, object)
194                         .doOnError(th -> LOGGER.error("Exception saving object asynchronously " + object, th))
195                         .blockingGet(object);
196                     LOGGER.debug("Saved object " + object + " as " + saved);
197                     return saved;
198                 } catch (Exception e) {
199                     LOGGER.error("Exception saving object " + object, e);
200                     throw e;
201                 }
202             }
203         }
204 
205         throw new IllegalStateException(
206             "Cannot find method to save " + object + ". Please, create a repository for type " + type
207         );
208     }

210     @Override
211     @SuppressWarnings({"rawtypes"})
212     public String getId(Class type, Map<String, Object> properties) {
213         return getIdName(type)
214             .flatMap(idName -> Optional.ofNullable(properties.get(idName)))
215             .map(String::valueOf)
216             .orElse(null);
217     }

219     @Override
220     @SuppressWarnings({"unchecked", "rawtypes"})
221     public <T> T addTo(T object, String association, Object other) {
222         ops().getEntity(object.getClass()).getPersistentProperties()
223             .stream()
224             .filter(p -> p.getName().equals(association))
225             .filter(p -> p instanceof RuntimeAssociation)
226             .findFirst()
227             .ifPresent(p -> {
228                 BeanProperty<T, ?> property = (BeanProperty<T, ?>) p.getProperty();
229                 Object collection = property.get(object);
230                 if (collection instanceof Collection) {
231                     Collection coll = (Collection) collection;
232                     coll.add(other);
233                 }
234             });
235 
236         return object;
237     }

243     @SuppressWarnings("unchecked")
244     private <T> Maybe<T> asMaybeEntity(Object result, T original) {
245         if (original.getClass().isInstance(result)) {
246             return Maybe.just((T) result);
247         }
248         if (result instanceof Optional) {
249             Optional<T> optional = (Optional<T>) result;
250             return optional.map(Maybe::just).orElseGet(Maybe::empty);
251         }
252         if (result instanceof CompletableFuture) {
253             return Maybe.create(emitter -> {
254                 CompletableFuture<T> future = (CompletableFuture<T>) result;
255                 try {
256                     emitter.onSuccess(future.get());
257                     emitter.onComplete();
258                 } catch (Throwable th) {
259                     if (th.getCause() instanceof EmptyResultException) {
260                         emitter.onComplete();
261                     } else {
262                         emitter.onError(th);
263                     }
264                 }
265             });
266         }
267         if (result instanceof Single) {
268             return Maybe.fromSingle((Single<T>) result);
269         }
270         if (result instanceof Maybe) {
271             return (Maybe<T>) result;
272         }
273         if (result instanceof Publisher) {
274             Publisher<T> publisher = (Publisher<T>) result;
275             return Maybe.create(emitter -> publisher.subscribe(new Subscriber<T>() {
276                 @Override
277                 public void onSubscribe(Subscription s) {
278                     s.request(1);
279                 }
280 
281                 @Override
282                 public void onNext(T t) {
283                     emitter.onSuccess(t);
284                 }
285 
286                 @Override
287                 public void onError(Throwable t) {
288                     emitter.onError(t);
289                 }
290 
291                 @Override
292                 public void onComplete() {
293                     emitter.onComplete();
294                 }
295             }));
296         }
297 
298         LOGGER.warn("Cannot cast object " + result + " to Maybe<" + original.getClass() + ">");
299 
300         return Maybe.empty();
301     }

303     @SuppressWarnings("unchecked")
304     private <T> Object getOriginalId(T object) {
305         BeanIntrospection<T> introspection = (BeanIntrospection<T>) ops().getEntity(object.getClass()).getIntrospection();
306         return getIdName(object.getClass())
307             .flatMap(introspection::getProperty)
308             .map(p -> p.get(object))
309             .orElse(null);
310     }
311 
312     @SuppressWarnings({"rawtypes", "unchecked"})
313     public Optional<String> getIdName(Class type) {
314         RuntimePersistentProperty identity = ops().getEntity(type).getIdentity();
315         if (identity == null) {
316             return Optional.empty();
317         }
318 
319         return Optional.of(identity.getName());
320     }
321 
322     @SuppressWarnings({"rawtypes", "unchecked"})
323     public void removeId(Object object) {
324         RuntimePersistentProperty identity = ops().getEntity(object.getClass()).getIdentity();
325         if (identity == null) {
326             return;
327         }
328 
329         BeanProperty property = identity.getProperty();
330         if (!property.isReadOnly()) {
331             property.set(object, null);
332         }
333     }

335     @SuppressWarnings("rawtypes")
336     private Optional<ExecutableMethod<GenericRepository, ?>> tryFindFindByIdMethod(Class<?> type, BeanDefinition<GenericRepository> definition) {
337         return tryFindMethod(type, definition, m ->
338             FIND_BY_ID_METHOD_NAME.equals(m.getMethodName())
339                 || getIdName(type).map(name -> (FIND_BY_ID_METHOD_NAME_PREFIX + StringUtils.capitalize(name)).equals(m.getMethodName())).orElse(false)
340         );
341     }
342 
343     @SuppressWarnings("rawtypes")
344     private Optional<ExecutableMethod<GenericRepository, ?>> tryFindSaveMethod(Class<?> type, BeanDefinition<GenericRepository> definition) {
345         return tryFindMethod(type, definition, m ->
346             PERSIST_METHOD_NAMES.stream().anyMatch(prefix -> m.getMethodName().startsWith(prefix))
347                 && m.getArgumentTypes().length == 1
348                 && m.getArgumentTypes()[0].equals(type)
349         );
350     }
351 
352     @SuppressWarnings("rawtypes")
353     private Optional<ExecutableMethod<GenericRepository, ?>> tryFindUpdateMethod(Class<?> type, BeanDefinition<GenericRepository> definition) {
354         return tryFindMethod(type, definition, m ->
355             UPDATE_METHOD_NAME.equals(m.getMethodName())
356                 && m.getArgumentTypes().length == 1
357                 && m.getArgumentTypes()[0].equals(type)
358         );
359     }
360 
361     @SuppressWarnings("rawtypes")
362     private Optional<ExecutableMethod<GenericRepository, ?>> tryFindMethod(Class<?> type, BeanDefinition<GenericRepository> definition, Predicate<? super ExecutableMethod<GenericRepository, ?>> filter) {
363         return definition.getExecutableMethods().stream().filter(filter).findFirst();
364     }
