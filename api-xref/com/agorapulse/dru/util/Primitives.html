18  package com.agorapulse.dru.util;
19  
20  import java.util.*;
21  
22  /**
23   * Contains static utility methods pertaining to primitive types and their corresponding wrapper
24   * types.
25   *
26   * Copied from Guava to prevent unnecessary dependency.
27   *
28   * @author Kevin Bourrillion
29   * @since 1.0
30   */
31  public final class Primitives {
32    private Primitives() {}
33  
34    /** A map from primitive types to their corresponding wrapper types. */
35    private static final Map<Class<?>, Class<?>> PRIMITIVE_TO_WRAPPER_TYPE;
36  
37    /** A map from wrapper types to their corresponding primitive types. */
38    private static final Map<Class<?>, Class<?>> WRAPPER_TO_PRIMITIVE_TYPE;
39  
40    // Sad that we can't use a BiMap. :(
41  
42    static {
43      Map<Class<?>, Class<?>> primToWrap = new LinkedHashMap<>(16);
44      Map<Class<?>, Class<?>> wrapToPrim = new LinkedHashMap<>(16);
45  
46      add(primToWrap, wrapToPrim, boolean.class, Boolean.class);
47      add(primToWrap, wrapToPrim, byte.class, Byte.class);
48      add(primToWrap, wrapToPrim, char.class, Character.class);
49      add(primToWrap, wrapToPrim, double.class, Double.class);
50      add(primToWrap, wrapToPrim, float.class, Float.class);
51      add(primToWrap, wrapToPrim, int.class, Integer.class);
52      add(primToWrap, wrapToPrim, long.class, Long.class);
53      add(primToWrap, wrapToPrim, short.class, Short.class);
54      add(primToWrap, wrapToPrim, void.class, Void.class);
55  
56      PRIMITIVE_TO_WRAPPER_TYPE = Collections.unmodifiableMap(primToWrap);
57      WRAPPER_TO_PRIMITIVE_TYPE = Collections.unmodifiableMap(wrapToPrim);
58    }
59  
60    private static void add(
61        Map<Class<?>, Class<?>> forward,
62        Map<Class<?>, Class<?>> backward,
63        Class<?> key,
64        Class<?> value) {
65      forward.put(key, value);
66      backward.put(value, key);
67    }
68  
69    /**
70     * Returns an immutable set of all nine primitive types (including {@code void}). Note that a
71     * simpler way to test whether a {@code Class} instance is a member of this set is to call {@link
72     * Class#isPrimitive}.
73     *
74     * @since 3.0
75     */
76    public static Set<Class<?>> allPrimitiveTypes() {
77      return PRIMITIVE_TO_WRAPPER_TYPE.keySet();
78    }
79  
80    /**
81     * Returns an immutable set of all nine primitive-wrapper types (including {@link Void}).
82     *
83     * @since 3.0
84     */
85    public static Set<Class<?>> allWrapperTypes() {
86      return WRAPPER_TO_PRIMITIVE_TYPE.keySet();
87    }
88  
89    /**
90     * Returns {@code true} if {@code type} is one of the nine primitive-wrapper types, such as {@link
91     * Integer}.
92     *
93     * @see Class#isPrimitive
94     */
95    public static boolean isWrapperType(Class<?> type) {
96      return WRAPPER_TO_PRIMITIVE_TYPE.containsKey(Objects.requireNonNull((type)));
97    }
98  
99    /**
100    * Returns the corresponding wrapper type of {@code type} if it is a primitive type; otherwise
101    * returns {@code type} itself. Idempotent.
102    *
103    * <pre>
104    *     wrap(int.class) == Integer.class
105    *     wrap(Integer.class) == Integer.class
106    *     wrap(String.class) == String.class
107    * </pre>
108    */
109   public static <T> Class<T> wrap(Class<T> type) {
110     Objects.requireNonNull(type);
111 
112     // cast is safe: long.class and Long.class are both of type Class<Long>
113     @SuppressWarnings("unchecked")
114     Class<T> wrapped = (Class<T>) PRIMITIVE_TO_WRAPPER_TYPE.get(type);
115     return (wrapped == null) ? type : wrapped;
116   }
117 
118   /**
119    * Returns the corresponding primitive type of {@code type} if it is a wrapper type; otherwise
120    * returns {@code type} itself. Idempotent.
121    *
122    * <pre>
123    *     unwrap(Integer.class) == int.class
124    *     unwrap(int.class) == int.class
125    *     unwrap(String.class) == String.class
126    * </pre>
127    */
128   public static <T> Class<T> unwrap(Class<T> type) {
129       Objects.requireNonNull(type);
130 
131     // cast is safe: long.class and Long.class are both of type Class<Long>
132     @SuppressWarnings("unchecked")
133     Class<T> unwrapped = (Class<T>) WRAPPER_TO_PRIMITIVE_TYPE.get(type);
134     return (unwrapped == null) ? type : unwrapped;
135   }
136 }
