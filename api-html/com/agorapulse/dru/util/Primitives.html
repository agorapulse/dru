018 package com.agorapulse.dru.util;
019 
020 import java.util.*;
021 
022 /**
023  * Contains static utility methods pertaining to primitive types and their corresponding wrapper
024  * types.
025  *
026  * Copied from Guava to prevent unnecessary dependency.
027  *
028  * @author Kevin Bourrillion
029  * @since 1.0
030  */
031 public final class Primitives {
032   private Primitives() {}
033 
034   /** A map from primitive types to their corresponding wrapper types. */
035   private static final Map<Class<?>, Class<?>> PRIMITIVE_TO_WRAPPER_TYPE;
036 
037   /** A map from wrapper types to their corresponding primitive types. */
038   private static final Map<Class<?>, Class<?>> WRAPPER_TO_PRIMITIVE_TYPE;
039 
040   // Sad that we can't use a BiMap. :(
041 
042   static {
043     Map<Class<?>, Class<?>> primToWrap = new LinkedHashMap<>(16);
044     Map<Class<?>, Class<?>> wrapToPrim = new LinkedHashMap<>(16);
045 
046     add(primToWrap, wrapToPrim, boolean.class, Boolean.class);
047     add(primToWrap, wrapToPrim, byte.class, Byte.class);
048     add(primToWrap, wrapToPrim, char.class, Character.class);
049     add(primToWrap, wrapToPrim, double.class, Double.class);
050     add(primToWrap, wrapToPrim, float.class, Float.class);
051     add(primToWrap, wrapToPrim, int.class, Integer.class);
052     add(primToWrap, wrapToPrim, long.class, Long.class);
053     add(primToWrap, wrapToPrim, short.class, Short.class);
054     add(primToWrap, wrapToPrim, void.class, Void.class);
055 
056     PRIMITIVE_TO_WRAPPER_TYPE = Collections.unmodifiableMap(primToWrap);
057     WRAPPER_TO_PRIMITIVE_TYPE = Collections.unmodifiableMap(wrapToPrim);
058   }
059 
060   private static void add(
061       Map<Class<?>, Class<?>> forward,
062       Map<Class<?>, Class<?>> backward,
063       Class<?> key,
064       Class<?> value) {
065     forward.put(key, value);
066     backward.put(value, key);
067   }
068 
069   /**
070    * Returns an immutable set of all nine primitive types (including {@code void}). Note that a
071    * simpler way to test whether a {@code Class} instance is a member of this set is to call {@link
072    * Class#isPrimitive}.
073    *
074    * @since 3.0
075    */
076   public static Set<Class<?>> allPrimitiveTypes() {
077     return PRIMITIVE_TO_WRAPPER_TYPE.keySet();
078   }
079 
080   /**
081    * Returns an immutable set of all nine primitive-wrapper types (including {@link Void}).
082    *
083    * @since 3.0
084    */
085   public static Set<Class<?>> allWrapperTypes() {
086     return WRAPPER_TO_PRIMITIVE_TYPE.keySet();
087   }
088 
089   /**
090    * Returns {@code true} if {@code type} is one of the nine primitive-wrapper types, such as {@link
091    * Integer}.
092    *
093    * @see Class#isPrimitive
094    */
095   public static boolean isWrapperType(Class<?> type) {
096     return WRAPPER_TO_PRIMITIVE_TYPE.containsKey(Objects.requireNonNull((type)));
097   }
098 
099   /**
100    * Returns the corresponding wrapper type of {@code type} if it is a primitive type; otherwise
101    * returns {@code type} itself. Idempotent.
102    *
103    * <pre>
104    *     wrap(int.class) == Integer.class
105    *     wrap(Integer.class) == Integer.class
106    *     wrap(String.class) == String.class
107    * </pre>
108    */
109   public static <T> Class<T> wrap(Class<T> type) {
110     Objects.requireNonNull(type);
111 
112     // cast is safe: long.class and Long.class are both of type Class<Long>
113     @SuppressWarnings("unchecked")
114     Class<T> wrapped = (Class<T>) PRIMITIVE_TO_WRAPPER_TYPE.get(type);
115     return (wrapped == null) ? type : wrapped;
116   }
117 
118   /**
119    * Returns the corresponding primitive type of {@code type} if it is a wrapper type; otherwise
120    * returns {@code type} itself. Idempotent.
121    *
122    * <pre>
123    *     unwrap(Integer.class) == int.class
124    *     unwrap(int.class) == int.class
125    *     unwrap(String.class) == String.class
126    * </pre>
127    */
128   public static <T> Class<T> unwrap(Class<T> type) {
129       Objects.requireNonNull(type);
130 
131     // cast is safe: long.class and Long.class are both of type Class<Long>
132     @SuppressWarnings("unchecked")
133     Class<T> unwrapped = (Class<T>) WRAPPER_TO_PRIMITIVE_TYPE.get(type);
134     return (unwrapped == null) ? type : unwrapped;
135   }
136 }
